func main() {
	var numero1 float64
	var numero2 float64
	var numero3 float64
	fmt.Print("Insira o primeiro número: ")
	fmt.Scan(&numero1)
	fmt.Print("Insira o segundo número: ")
	fmt.Scan(&numero2)
	fmt.Print("Insira o terceiro número: ")
	fmt.Scan(&numero3)
	var resultado float64 = ((numero1 * 2) + (numero2 * 3) + (numero3 * 5)) / 10
	fmt.Print("Sua média ponderada é: ", resultado)

}
